  • 重复的从待排元素中选出最大者或 最小者
    • 我之前一般都是存元素，这里存的是下标
void SelectSort(vector<int> nums){
    int len = nums.size();
    for(int i=0;i<len;i++){
        //将a[i]与a[i+1...len]中最小元素交换
        int min = i;//最小元素的索引
        for(int j=i+1;j<len;j++){
            if(less(a[j],a[min]))//这个less的写法，自己要会,lionel
                min=j;
        }
        exch(a,i,min);
    }
}
